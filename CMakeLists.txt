function(add_symlink source_file link_name)
    if (EXISTS ${source_file})
        file(CREATE_LINK ${source_file} ${link_name} SYMBOLIC)
    else()
        message(WARNING "${source_file} not found. Skipping symlink creation for ${link_name}.")
    endif()
endfunction()
